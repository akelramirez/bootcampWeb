//Encapsulamiento
function Bootcamp (clase){
    let Topico = "Web Development"; //variable privada
    this.clase = clase;

    this.getTopico = function(){
        return Topico;
    }; // Método publico para acceder a la variable privada

    this.setTopico = function(nuevoTopico){
        if (nuevoTopico != null){
           Topico = nuevoTopico; //Método público para modificar la variable privada
        };
    };
}
